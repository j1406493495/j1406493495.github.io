def bsearch(array, n, value):
    low = 0
    high = n - 1

    while (low <= high):
        # mid = (low + high) >> 1, 防止两数相加溢出
        mid = low + ((high - low) >> 1)

        if (array[mid] < value):
            low = mid + 1
        elif (array[mid] > value):
            high = mid - 1
        elif (array[mid] == value):
            # if (array[mid + 1] > array[mid]): ,添加是否最后一个元素判断，防止index溢出
            if ((mid == n - 1) or (array[mid + 1] > array[mid])):
                return mid
            else:
                low = mid + 1


result = bsearch([1, 3, 4, 4, 4, 6, 7, 9], 8, 9)
print(str(result))
